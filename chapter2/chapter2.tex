\section{Microstate and Macrostate}\label{sec:microstate_and_macrostate} % (fold)
In the first chapter, we used an ideal statistical model to develop the kinetic theory of gases and obtained some very useful results that formed the basis of the first law of thermodynamics. Now, we continue to use our knowledge of probability and statistics to answer our question: "Why does heat\footnote[2]{Heat in this context, is a \textbf{positive quantity.}} flow from high to low temperatures, but not in the reverse direction?".
\\ \\ \\ \indent Let's do some combinatorics here. Imagine you have a fair coin, and you flip that coin $3$ times. You define an event $E$ as "Landing $2$ heads". How many sample points are there inside the subset $E$?
$$ E=\{THH,HTH,THH\} \Rightarrow n(E)=\binom{3}{2}=3$$
How many possible outcomes are there inside the sample space $\Omega$ (or $S$)?
$$ \Omega=\{TTT,TTH,THT,THH,HHH,HHT,HTH,HHH\}\Rightarrow n(\Omega)=2^{3}=8$$
The event $E$ is called as the \textbf{macrostate} of the system, and each of its elements is called \textbf{microstate}. For example, the macrostate "Landing $2$ heads" has $3$ microstates, one of them is $THH$; the number of microstates corresponding to a macrostate is called the \textbf{multiplicity}. Now I will change the notation here, I use the symbol $\Omega$ to denote \textbf{multiplicity}\footnote[3]{You can relate this new convention to the \textbf{random variable} concept in math books.}.
Generally, the multiplicity of the macrostate "Landing $n$ heads" when flipping a coin $N$ times is:
\begin{equation*}
	\Omega(N,n)=\binom{N}{n}
\end{equation*}

A solid contains $N$ chemical bonds\footnote[1]{I think it is more accurate to use term "chemical bonds", rather than "oscillators" here.}\footnote[2]{As of the time of writing, I still do not understand why solids have $6$ degrees of freedom. Atoms in a crystal lattice barely move, so logically, their kinetic energy should be zero, and $f=3$ is simply due to potential energy stored inside the ionic bonds?}, and store a total of $q$ units of energy. The multiplicity is simply the number of \textbf{non-negative} solutions of this equation:
\begin{equation*}
	x_{1}+x_{2}+\cdots+x_{N-1}+x_{N}=q
\end{equation*}
Or equivalently, the number of each solution must be \textbf{greater than or equal to $1$} of this equation:
\begin{equation*}
	(x_{1}+1)+(x_{2}+1)+\cdots +(x_{N-1}+1)+(x_{N}+1)=q+N
\end{equation*}
\begin{equation*}
	\Leftrightarrow t_{1}+t_{2}+\cdots+t_{N-1}+t_{N}=q+N
\end{equation*}
This combinatorial problem is very classic; we just need to "split" $(q+N)$ units of energy using $(N-1)$ "bars".

This crucial formula is the backbone of this section, especially when we deal with problems related to Einstein's solid model.
\begin{theorem}[The Multiplicity of an Einstein solid]
	The general formula for the multiplicity of an Einstein solid with $N$ chemical bonds and $q$ units of energy is:
	\begin{equation*}
		\Omega(N,q)=\binom{N+q-1}{q}=\frac{(N+q-1)! }{(N-1)!q!}
	\end{equation*}
	\label{The Multiplicity of an Einstein solid}
\end{theorem}
In our previous example in Figure ~\ref{fig:2_1}, the number of satisfied configurations is:
$$ \Omega(N,q)=\Omega(3,4)=\binom{3+4-1}{4}=15 $$
The problem is, $N$ and $q$ are always very large numbers. For example, in $1$ gram of $\ce{Na}$, the number of chemical bonds is up to $N=7.856.10^{22}$. In statistical physics, we usually deal with very large number; therefore, in the remainder of this subsection, I will introduce some powerful mathematical tools to handle them easier.
\begin{theorem}[Weak Stirling's Approximation for Natural Logarithm] When $N>>1$:
	$$\ln{N!}=N\ln{N}-N $$
	\label{Weak Stirling's Approximation}
\end{theorem}
\begin{proof}
	\begin{equation*}
		\begin{split}
			\ln{N!}&=\ln{[N(N-1)\cdots 2.1]}=\ln{N}+\ln{(N-1)}+\cdots+\ln{2}+\ln{1}\\
			&=\sum_{i=1}^{N}\ln{i}\approx\int_{1}^{N}\ln{x}dx\approx N\ln{N}-N
		\end{split}
	\end{equation*}

\end{proof}
\begin{theorem}[Stirling's Approximation] When $N>>1$:
	$$    N!\approx N^{N}e^{-N}\sqrt{2\pi N}$$
\end{theorem}

\begin{proof}
	Recall the definition of Gamma function, for every non-negative integer $N$, we always have:
	\begin{equation*}
		\begin{split}
			\Gamma(N+1)&=\int_{0 }^{+\infty}x^{N}e^{-x}dx=N!\\
			&=\int_{0 }^{+\infty} e^{N\ln{x}}e^{-x}dx=\int_{0 }^{+\infty} e^{N\ln{x}-x}dx\\
			&=\int_{0 }^{+\infty}e^{N\ln{x}-x}dx
		\end{split}
	\end{equation*}
	Now we want to expand the equation by Taylor expansion, by setting an auxiliary variable $x=y+N\Rightarrow y=x-N$ :
	\begin{equation*}
		\begin{split}
			N\ln{x}-x&=N\ln{(y+N)}-(y+N)\\
			&=N\ln{N\left(\frac{y }{N}+1\right)}-(y+N)\\
			&= N\ln{N}+N \left(\frac{y }{N}-\frac{1 }{2}.\frac{y^{2} }{N^{2}}\right)-y-N\\
			&=N\ln{N}-\frac{1 }{2}.\frac{y^{2} }{N}-N\\
		\end{split}
	\end{equation*}
	After changing the variable, we obtain:
	\begin{equation*}
		\begin{split}
			\Gamma(N+1)&=\int_{0 }^{+\infty}e^{N\ln{x}-x}dx=\int_{-N }^{+\infty}\exp{\left(N\ln{N}-\frac{1 }{2}.\frac{y^{2} }{N}-N\right)} dy\\
			&=N^{N}e^{-N}\int_{-N }^{+\infty}\exp \left(-\frac{1 }{2}.\frac{y^{2}}{N}\right) dy
		\end{split}
	\end{equation*}
	Since $N$ approaches infinity, applying Gaussian integral result:
	$$ \int_{-\infty }^{+\infty }e^{-y^{2}}dy=\sqrt{\pi}  $$
	We conclude:
	\begin{equation*}
		\begin{split}
			\Gamma(N+1)&=N^{N}e^{-N}\sqrt{2N}\int_{-\infty }^{+\infty}\exp \left[-\left(\frac{y }{\sqrt{2N}}\right)^{2}\right] d \left(\frac{y }{\sqrt{2N}}\right)\\
			&=N^{N}e^{-N}\sqrt{2\pi N}
		\end{split}
	\end{equation*}
\end{proof}

\begin{corollary}[Multiplicity of a Large Einstein Solid]
	\label{Multiplicity of a Large Einstein Solid}
	The multiplicity of a large Einstein solid at normal temperature ($q>>N$) is:
	\begin{equation*}
		\ln{\Omega(N,q)}=N+N\ln{\frac{q }{N}}\Rightarrow \Omega(N,q)=\left(\frac{eq }{N}\right)^{N}
	\end{equation*}
\end{corollary}
\begin{proof}
	\begin{equation*}
		\begin{split}
			\ln{\Omega(N,q)}&=(q+N)\ln{(q+N)}-q\ln{q}-N\ln{N}\\
			&=(q+N)\ln{\left[q \left(\frac{N}{q}+1\right)\right]}-q\ln{q}-N\ln{N}\\
			&\approx (q+N)\left(\ln{q}+\frac{N }{q}\right)-q\ln{q}-N\ln{N}\\
			&=N\ln{\frac{q }{N}}+N+\frac{N^{2}}{q}\\
			&\approx N\ln{\frac{q }{N}}+N
		\end{split}
	\end{equation*}
	We can see that:
	\begin{equation*}
		\Omega(N,q)=\left(\frac{eq }{N}\right)^{N}
	\end{equation*}
\end{proof}

If we assume the initial internal energy of solid $A$ is $q_{A}=0$ and solid $B$ is $q_{B}=q$, then very quickly the energy is \textbf{well-distributed} due to multiplicities. Solid $A$ receives the energy from solid $B$, and solid $B$ transfers its energy to solid $A$; until they both reach the equilibrium state. Since internal energy is directly proportional to temperature, heat will be transferred from the higher temperature object (solid $B$) to the lower temperature object (solid $A$). This is a very convincing answer for our previous question from previous chapter!
\begin{theorem}[The Second Law of Thermodynamics]
	\label{The Second Law of Thermodynamics}
	Multiplicity tends to increase; and any large system in equilibrium will be found in the macrostate with the greatest multiplicity.
\end{theorem}
\begin{corollary}[Heat flowing direction]
	\textbf{Statistically}, heat flows from the higher temperature object to the lower temperature object, and never in reverse.
	\label{Heat flowing direction}
\end{corollary}
As the examples above have shown, when $(q, N)$ are very large numbers, it is more convenient to deal with $\ln\Omega\footnote[2]{From now on, $\Omega$ is denoted as the short form for $\Omega(N,q)$.}$ than $\Omega$ itself. Now we define a new quantity, called \textbf{entropy}, to scale the multiplicity $\ln\Omega$ down, and assign it with a physical unit.
\begin{definition}[Entropy Definition]
	Entropy is defined as: $$ S=k_{B}\ln\Omega \quad(\text{J/K})$$
\end{definition}
In this chapter, we still do not explain why $k_{B}$ is chosen to be scale factor yet, or physical meaning of the quantity $S$. Keep everything as simple as possible, and consider the quantity \textbf{entropy} as another representation of \textbf{multiplicity}. From Theorem ~\ref{The Second Law of Thermodynamics}, we can see that:
\begin{corollary}
	Entropy tends to increase.
\end{corollary}
To conclude this section, we now develop some useful results in calculating entropy of a general Einstein solid.
\subsection{$N$ and $q$ are relatively small}
If both $N$ and $q$ values are small, you can calculate exactly entropy of an Einstein solid without using any approximation formula.
\begin{equation*}
	S=k_{B}\ln{\Omega}=k_{B}\ln \binom{N+q-1}{q}
\end{equation*}
\subsection{$N$ and $q$ are large values}\label{sub:_n_and_q_are_large_values} % (fold)
\subsubsection{An Einstein solid at high-temperature}
As we discussed above; if a solid is at high-temperature compared to absolute zero, then $q>>N$. We have proved the approximation formula:
$$ \Omega\approx \left(\frac{eq}{N}\right)^{N} $$
Now we can conclude: $$ S=k_{B}\ln{\Omega}=k_{B}N\ln{\frac{eq }{N}} $$
% subsection $N$ and $q$ are large values (end)
\subsubsection{An Einstein solid at low-temperature}
At low temperature near absolute zero, intuitively $N>>q$, we have to develop another approximation formula for this case. Since this is not a typical case, we will not state any corollaries here.
\begin{equation*}
	\begin{split}
		\ln\Omega&=(q+N)\ln{(q+N)}-q\ln{q}-N\ln{N}\\
		&=(q+N)\ln\left[N \left(\frac{q }{N}+1\right)\right]-q\ln{q}-N\ln{N}\\
		&\approx (q+N)\ln{N}+(q+N)\frac{q }{N}-q\ln{q}-N\ln{N}\\
		&=q\ln{N}+\frac{q^{2}}{N}+q-q\ln{q}\\
		&\approx q\ln{\frac{N }{q}}+q=\ln \left(\frac{eN}{q}\right)^{q}\\
	\end{split}
\end{equation*}
Finally, we see that: $$ S=k_{B}\ln{\Omega}=k_{B}q\ln \frac{eN }{q} $$
\subsubsection{A general Einstein solid model}\label{sec:a_general_einstein_solid} % (fold)
Now we are considering the final case, which both $N$ and $q$ are large value; and their difference is \textbf{not significant}. For example, assume that $N=2.31.10^{5}$ and $q=2.43.10^{5}$, obviously we must derive our approximation from scratch because the initial conditions $q>>N$ or $N>>q$ are incorrect.
\begin{equation*}
	\begin{split}
		\Omega&=\binom{N+q-1}{q}=\frac{(N+q-1)! }{q!(N-1)!}=\frac{N }{q+N}\frac{(q+N)! }{q!N!}\\
		&=\frac{N }{q+N}\frac{\sqrt{2\pi(q+N)}\left[(q+N)/e\right]^{q+N}}{\sqrt{2\pi q}\left(q/e\right)^{q}\sqrt{2\pi N}(N/e)^{N}}\quad(\text{Stirling's approximation})\\
		&=\frac{\sqrt{N}}{\sqrt{2\pi}\sqrt{q(q+N)}}\left(\frac{q+N }{q}\right)^{q}\left(\frac{q+N }{N}\right)^{N}\\
		&= \frac{\left(\displaystyle\frac{q+N}{q}\right)^{q}\left(\displaystyle\frac{q+N}{N}\right)^{N} }{\sqrt{\displaystyle\frac{2\pi q(q+N)}{N}}}
	\end{split}
\end{equation*}
For any large values of $N$ and $q$, the entropy of Einstein solid is:
\begin{equation*}
	S=k_{B}\ln\Omega=k_{B}\ln\left[\frac{\left(\displaystyle\frac{q+N}{q}\right)^{q}\left(\displaystyle\frac{q+N}{N}\right)^{N} }{\sqrt{\displaystyle\frac{2\pi q(q+N)}{N}}}
		\right]
\end{equation*}

\section{The Ideal Gas Model}\label{sec:the_ideal_gas_model} % (fold)
\subsection{Fourier Transform and Quantum Mechanics}
Every energy signals in time domain can be analyzed in frequency domain through Fourier transform\footnote[1]{I think the most convenient way to represent the Fourier transform is complex exponential form, which I learned from Signals and Systems course in my university. You should refer to Chapter 3, "Signals and Systems", Alan V. Oppenheim for the full proof of these formulas.}\footnote[2]{If this is your first time encountering the Fourier transform, I highly recommend watching \href{https://youtu.be/PRUn5Alm6Vk?si=fncmV6eVDB1_hrNu}{this video} to grasp the basic idea behind it.}:
$$ X(\omega)=\int_{-\infty }^{+\infty}x(t)e^{-j\omega t}dt=\mathcal{F}(x(t))  $$
Similarly, spectrum in frequency domain can also be reconstructed to signal in time domain through inverse Fourier transform:
$$ x(t)=\frac{1}{2\pi}\int_{-\infty }^{+\infty}X(\omega)e^{j\omega t}d\omega=\mathcal{F}^{-1}(X(\omega))  $$
Now let's build a bridge. Imagine a single gas particle inside a box of volume $V$; it moves from point $A$ to point $B$. All trajectories are in "position space", and its momentum is in "momentum space". Loosely speaking, if you know the particle's path from $A$ to $B$, then you can relatively know its movement. Similarly, the transformation between volume space and momentum space can also be described as:
$$ \phi(p)=\frac{1 }{\sqrt{2\pi\hbar}}\int_{-\infty }^{+\infty}\psi(x)\exp{\left({\frac{-jpx }{\hbar}}\right)}dx  $$

$$ \psi(x)=\frac{1 }{\sqrt{2\pi\hbar}}\int_{-\infty }^{+\infty}\phi(p)\exp{\left({\frac{jpx }{\hbar}}\right)}dp  $$
Let's take a look at these 2 formulas, you can see many similarities here. First, both frequency domain and momentum space are imaginary; conversely, both time domain and position space are real. Second, neglecting all constants, the Fourier transform and "position-momentum" transform are consistent. These quantities can be loosely related in physical meaning like this:
\begin{equation*}
	\begin{split}
		x(t)\text{ (Time domain)}&\leftrightarrow \psi(x)\text{ (Position space)}\\
		X(\omega)\text{ (Frequency domain)}&\leftrightarrow \phi(p)\text{ (Momentum space)}\\
	\end{split}
\end{equation*}

	\end{center}

Why do I know the momentum space of a single gas particle is a sphere? Generally, the internal energy of a system is $U$, we have:
$$ U=\frac{1 }{2}m(v_{x}^{2}+v_{y}^{2}+v_{z}^{2})=\frac{1 }{2m}(p_{x}^{2}+p_{y}^{2}+p_{z}^{2}) $$
Equivalently, it yields:
$$ p_{x}^{2}+p_{y}^{2}+p_{z}^{2}=2mU \text{ (Sphere equation)} $$
Notice that our momentum space $\phi(p)$ is just \textbf{the surface area} of the sphere. Now we move on to Heisenberg's uncertainty principle. Let's take a look at 2 signals and their Fourier representations.

		\includegraphics[width=0.7\textwidth]{~/PhyAI/chapter2/2_5.png}
	\end{center}
	\caption{$\delta(t)$ and $\cos(\omega_{0}t$)}\label{fig:2_5}
\end{figure}
The Dirac delta $\delta(t)$ signal focuses only on the origin in the time domain, but its spectrum spreads out entirely in the frequency domain. The cosine $\cos(\omega_{0}t)$ signal spreads out over the time domain; but its spectrum focuses only on the frequency values $\omega=\omega_{0}$ and $\omega=-\omega_{0}$.
\\ \\
\textit{The more information you observe in the time domain, the less information you know in the frequency domain; and vice versa.}
\\ \\
Using the same analogy, for simplify, you can qualitatively think Heisenberg's uncertainty principle as:
\\ \\
\textit{The less spread out the wavefunction is in position space, the more spread out it must be in momentum space; and vice versa.}
\\ \\
This statement can be rewritten as: $$ (\Delta x)(\Delta p_{x})\approx h $$
where $\Delta x$ is the spread in $x$, $\Delta p_{x}$ is the spread in $p$, and $h$ is Planck's constant.
\begin{figure}[H]
	\begin{center}
		\includegraphics[width=0.8\textwidth]{~/PhyAI/chapter2/2_6.png}
	\end{center}
	\caption{A number of position and momentum states for particle moving along $x$ direction}\label{fig:2_6}
\end{figure}
The total number of distinct states on $x$ direction is: $$ \frac{L }{\Delta x}\frac{L_{p}}{\Delta p_{x}}=\frac{LL_{p}}{h} $$
\subsection{Entropy of Ideal Gas}
In the previous subsection, I introduced a frame of quantum mechanics, that needed to calculate the entropy of ideal gas. Let's consider the simplest model, only one single ideal gas atom is inside the container. The system has internal energy $U$ (mostly concentrates on the kinetic energy), and assume that the container's volume is $V$. How many microstates could the atom be in? We already know the number of available states on a single direction; using the same line of thinking, multiplicity on 3 directions $x,y$ and $z$ is:
$$ \Omega_{1}=\frac{VV_{p}}{h^{3}} $$
Notice that $V_{p}$ is just \textbf{the surface area} of momentum space (in this case, it is a sphere). Add one more atom in our container, now what is the multiplicity of the system? Our momentum space now is more complicated:
$$ p_{1x}^{2}+p_{1y}^{2}+p_{1z}^{2}+p_{2x}^{2}+p_{2y}^{2}+p_{2z}^{2}= 2mU$$
Obviously, this is not an ordinary surface equation; it defines the surface of a hypersphere in $6$-dimensional momentum space.
\begin{figure}[H]
	\begin{center}
		\includegraphics[width=0.8\textwidth]{~/PhyAI/chapter2/2_7.png}
	\end{center}
	\caption{Multiplicity of a system containing $2$ atoms}\label{2_7}
\end{figure}
Similarly, we can see that:
$$ \Omega_{2}=\frac{V}{h^{3}}.\frac{V }{h^{3}}\times V_{p}=\frac{V^{2}}{h^{6}}\times V_{p} $$
But what is $V_{p}$? In the case of a single atom, $V_{p}$ is the surface area of a sphere of radius $R=\sqrt{2mU}$. Is $V_{p}$ still the "surface area" of a hypersphere in $6$-dimensional space? The answer is yes, and we have a general formula for the "surface-area" of any $d$-dimensional hypersphere of radius $r$.
\begin{theorem}["Surface area" of $d$-dimensional hypersphere of radius $r$ formula]
	\begin{equation*}
		A_{d}(r)=\displaystyle\frac{2\pi^{d/2}}{\left(\displaystyle\frac{d }{2}-1\right)!}r^{d-1}
	\end{equation*}
\end{theorem}
\begin{proof}
	The simplest way to prove this result is to use induction. Rewriting in the form of Gamma function yields:
	$$ A_{d}(r)=\displaystyle\frac{2\pi^{d/2}}{\Gamma \left(\displaystyle\frac{d }{2}\right)}r^{d-1} $$
	After substituing $d=1,2,3$, we obtain:
	\begin{equation*}
		\begin{split}
			A_{1}(r)&=2 \quad\text{($2$ points bounding the line segment)}\\
			A_{2}{(r)}&=2\pi r\quad\text{(Circumference of a circle)}\\
			A_{3}{(r)}&=4\pi r^{2}
		\end{split}
	\end{equation*}
	We do not care about the trivial case $A_{1}(r)$, but rather with guessing a recurrence relationship through how $A_{3}(r)$ is established.
	\begin{figure}[H]
		\begin{center}
			\includegraphics[width=0.8\textwidth]{~/PhyAI/chapter2/2_8.png}
		\end{center}
		\caption{Derivation of $A_{3}(r)$ formula  }\label{2_8}
	\end{figure}
	It is obvious from the drawing that we have:
	$$ A_{3}(r)=\int_{0 }^{\pi}A_{2}(r\sin\theta)rd\theta=  \int_{0 }^{\pi}2\pi r\sin\theta.rd\theta=4\pi r^{2} $$
	The recurrence relation is pretty clear, we try to prove it is still true even $d>3$.
	$$ A_{d}(r)=\int_{0 }^{\pi}A_{d-1}(r\sin\theta)rd\theta  $$
	Assume that our formula still holds true up to $d-1$ dimension:
	$$ A_{d-1}(r)=\displaystyle\frac{2\pi^{(d-1)/2}}{\Gamma \left(\displaystyle\frac{d-1 }{2}\right)} r^{d-2}$$
	Plug the function above into our recurrence relation:
	\begin{equation*}
		\begin{split}
			A_{d}(r)&=\int_{0}^{\pi}A_{d-1}(r\sin\theta)rd\theta=\displaystyle\frac{2\pi^{(d-1)/2}}{\Gamma \left(\displaystyle\frac{d-1 }{2}\right)}r^{d-1}\int_{0 }^{\pi}\sin^{d-2}\theta d\theta   \\
		\end{split}
	\end{equation*}
	After separating the constant, now only the Wallis's integral left. To complete our proof, we have to show that:
	$$ \int_{0 }^{\pi}\sin^{d-2}\theta d\theta=\displaystyle\frac{\sqrt{\pi} \Gamma \left(\displaystyle\frac{d-1}{2}\right)}{\Gamma \left(\displaystyle\frac{d}{2}\right)}  $$
	Setting $d-2=n$, we reduce our equation to:
	$$ \int_{0 }^{\pi}\sin^{n}\theta d\theta=\displaystyle\frac{\sqrt{\pi}\Gamma \left(\displaystyle\frac{n}{2}+\displaystyle\frac{1}{2}\right)}{\Gamma\left(\displaystyle\frac{n}{2}+1\right)}  $$
	Before directly proving, let me recall the recursive property of Wallis's integral:
	\begin{equation*}
		\begin{split}
			W_{n}&=\int_{0 }^{\pi}\sin^{n}\theta d\theta\\&=\int_{0 }^{\pi}\sin^{n-2}\theta (1-\cos^{2}\theta)d\theta\\ &=\int_{0 }^{\pi}\sin^{n-2}\thetad\theta-\int_{0 }^{\pi}\sin^{n-2}\theta\cos^{2}\theta d\theta  \\
			&=\int_{0 }^{\pi}\sin^{n-2}\theta d\theta -\int_{0 }^{\pi}\cos\theta d \left(\frac{\sin^{n-1}\theta }{n-1}\right)\\
			&=\int_{0 }^{\pi}\sin^{n-2}\theta d\theta -\frac{1 }{n-1}\int_{0 }^{\pi}\sin^{n}\theta d\theta \\
			&= W_{n-2}-\frac{1 }{n-1}W_{n}\\
		\end{split}
	\end{equation*}
	Finally, we can conclude:
	$$ W_{n}=\frac{n-1 }{n}W_{n-2} $$
	If $n$ is an even integer, then:
	\begin{equation*}
		\begin{split}
			W_{n}&=\frac{n-1 }{n}.\frac{n-3 }{n-2}\cdots \frac{1 }{2}.W_{0}=\frac{(n-1)!!}{n!!}\pi\\
			\Gamma \left(\frac{n}{2}+\frac{1 }{2}\right)&=\frac{n-1 }{2}.\frac{n-3 }{2}\cdots \frac{1}{2}.\Gamma \left(\frac{1}{2}\right)=\frac{(n-1)!!}{2^{n/2}}\sqrt{\pi}\\
			\Gamma \left(\frac{n }{2}+1\right)&=\frac{n }{2}.\frac{n-2 }{2}\cdots \frac{2 }{2}.\Gamma(0)=\frac{n!! }{2^{n/2}}.1\\
		\end{split}
	\end{equation*}
	We completed our proof for $n=2k$ case. If $ n $ is an odd integer, then:
	\begin{equation*}
		\begin{split}
			W_{n}&=\frac{n-1 }{n}.\frac{n-3 }{n-2}\cdots \frac{2 }{2}.W_{1}=\frac{(n-1)!!}{n!!}.2\\
			\Gamma \left(\frac{n}{2}+\frac{1 }{2}\right)&=\frac{n-1 }{2}.\frac{n-3 }{2}\cdots \frac{2}{2}.\Gamma \left(0\right)=\frac{(n-1)!!}{2^{n/2}}.1\\
			\Gamma \left(\frac{n }{2}+1\right)&=\frac{n }{2}.\frac{n-2 }{2}\cdots \frac{2 }{2}.\Gamma \left(\frac{1 }{2}\right)=\frac{n!! }{2^{n/2}}\sqrt{\pi}\\
		\end{split}
	\end{equation*}
	We completed our proof for $n=2k+1$ case.
\end{proof}
